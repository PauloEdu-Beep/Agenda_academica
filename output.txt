# --- ConexÃ£o DB ---
def get_db_connection():
    try:
        conn = pymysql.connect(host=db_config['host'],
                             user=db_config['user'],
                             password=db_config['password'],
                             database=db_config['database'],

        return conn
    except pymysql.MySQLError as err:
        print(f"Erro DB: {err}")
        return None

# --- Decorator de Login ---
def login_required(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        if 'usuario_id' not in session:
            flash("Por favor, faÃ§a login para acessar.", "warning")
            return redirect(url_for('login'))
        return f(*args, **kwargs)
    return decorated_function

# --- Rotas de AutenticaÃ§Ã£o ---
@app.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        email = request.form['email']
        senha = request.form['senha']
        
        conn = get_db_connection()
        cursor = conn.cursor()
        # Nota: Em produÃ§Ã£o, use hash real (bcrypt/argon2). Aqui compara string simples conforme PDF.
        cursor.execute("SELECT * FROM Usuario WHERE email = %s AND senha_hash = %s", (email, senha))
        user = cursor.fetchone()
        cursor.close()
        conn.close()

        if user:
            session['usuario_id'] = user['usuario_id']
            session['nome'] = user['nome_completo']
            session['tipo'] = user['tipo_usuario']
            return redirect(url_for('dashboard'))
        else:
            flash('Email ou senha invÃ¡lidos', 'danger')
            
    return render_template('login.html')

@app.route('/logout')
def logout():
    session.clear()
    return redirect(url_for('login'))

@app.route('/matriculas', methods=['GET', 'POST'])
@login_required
def gerenciar_matriculas():
    # Apenas Alunos se matriculam ou Admins matriculam alunos. 
    # SimplificaÃ§Ã£o: O prÃ³prio aluno se matricula.
    if session['tipo'] != 'ALUNO':
         flash("Apenas alunos podem realizar matrÃ­culas nesta versÃ£o.", "info")
         return redirect(url_for('dashboard'))

    conn = get_db_connection()
    cursor = conn.cursor()

    if request.method == 'POST':
        disciplina_id = request.form['disciplina_id']
        semestre = "2024.1" # Hardcoded para exemplo
        
        try:
            cursor.execute("INSERT INTO Matricula (usuario_id, disciplina_id, semestre) VALUES (%s, %s, %s)",
                           (session['usuario_id'], disciplina_id, semestre))
            conn.commit()
            flash('MatrÃ­cula realizada com sucesso!', 'success')
        except pymysql.MySQLError as e:
            conn.rollback()
            flash(f'Erro na matrÃ­cula (jÃ¡ matriculado?): {e}', 'danger')

    # Listar disciplinas disponÃ­veis vs disciplinas jÃ¡ matriculadas
    cursor.execute("""
        SELECT d.*, 
            (SELECT COUNT(*) FROM Matricula m WHERE m.disciplina_id = d.disciplina_id AND m.usuario_id = %s) as matriculado
        FROM Disciplina d
    """, (session['usuario_id'],))
    disciplinas = cursor.fetchall()
    
    cursor.close()
    conn.close()
    return render_template('matriculas.html', disciplinas=disciplinas)

@app.route('/compromissos')
@login_required
def list_compromissos():
    conn = get_db_connection()
    cursor = conn.cursor()
    
    user_id = session['usuario_id']
    tipo = session['tipo']

    # LÃ³gica CrÃ­tica: O que cada um vÃª?
    query = ""
    params = ()

    if tipo == 'ALUNO':
        # Aluno vÃª: Seus compromissos pessoais OU Compromissos das turmas (Prof_Disc) onde ele tem MatrÃ­cula
        query = """
            SELECT c.*, d.nome_disciplina, u_prof.nome_completo as professor_nome
            FROM Compromisso c
            LEFT JOIN Disciplina d ON c.turma_disciplina_id = d.disciplina_id
            LEFT JOIN Usuario u_prof ON c.turma_usuario_id = u_prof.usuario_id
            LEFT JOIN Matricula m ON m.disciplina_id = c.turma_disciplina_id AND m.usuario_id = %s
            WHERE c.aluno_usuario_id = %s 
               OR (c.turma_usuario_id IS NOT NULL AND m.usuario_id IS NOT NULL)
            ORDER BY c.data_hora_inicio ASC
        """
        params = (user_id, user_id)
        
    elif tipo == 'PROFESSOR':
        # Professor vÃª: Compromissos que ELE criou para suas turmas
        query = """
            SELECT c.*, d.nome_disciplina
            FROM Compromisso c
            LEFT JOIN Disciplina d ON c.turma_disciplina_id = d.disciplina_id
            WHERE c.turma_usuario_id = %s
            ORDER BY c.data_hora_inicio ASC
        """
        params = (user_id,)
        
    else: # ADMIN
        query = "SELECT * FROM Compromisso ORDER BY data_hora_inicio DESC"
        params = ()

    cursor.execute(query, params)
    compromissos = cursor.fetchall()
    cursor.close()
    conn.close()
    return render_template('compromissos.html', compromissos=compromissos)

@app.route('/compromissos/add', methods=['GET', 'POST'])
@login_required
def add_compromisso():
    conn = get_db_connection()
    cursor = conn.cursor()

    if request.method == 'POST':
        titulo = request.form['titulo']
        descricao = request.form['descricao']
        inicio = request.form['data_hora_inicio']
        fim = request.form.get('data_hora_fim') or None # Novo campo PDF
        tipo_comp = request.form['tipo_compromisso']
        
        # DefiniÃ§Ã£o de escopo baseada em quem estÃ¡ logado
        aluno_id = None
        turma_prof_id = None
        turma_disc_id = None

        if session['tipo'] == 'ALUNO':
            aluno_id = session['usuario_id'] # Compromisso Pessoal
        elif session['tipo'] == 'PROFESSOR':
            # Professor seleciona para qual turma Ã© o compromisso
            turma_selection = request.form.get('turma_id') # formato "prof_id-disc_id"
            if turma_selection:
                turma_prof_id, turma_disc_id = turma_selection.split('-')

        try:
            cursor.execute("""
                INSERT INTO Compromisso 
                (titulo, descricao, data_hora_inicio, data_hora_fim, tipo_compromisso, aluno_usuario_id, turma_usuario_id, turma_disciplina_id)
                VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
            """, (titulo, descricao, inicio, fim, tipo_comp, aluno_id, turma_prof_id, turma_disc_id))
            conn.commit()
            flash('Compromisso agendado!', 'success')
        except pymysql.MySQLError as err:
            flash(f'Erro: {err}', 'danger')
        
        cursor.close()
        conn.close()
        return redirect(url_for('list_compromissos'))

    # Preparar dados para o formulÃ¡rio
    turmas_professor = []
    if session['tipo'] == 'PROFESSOR':
        # Buscar turmas que este professor leciona (Professor_Disciplina)
        cursor.execute("""
            SELECT pd.usuario_id, pd.disciplina_id, d.nome_disciplina
            FROM Professor_Disciplina pd
            JOIN Disciplina d ON pd.disciplina_id = d.disciplina_id
            WHERE pd.usuario_id = %s
        """, (session['usuario_id'],))
        turmas_professor = cursor.fetchall()

    cursor.close()
    conn.close()
    return render_template('compromisso_form.html', turmas=turmas_professor)
